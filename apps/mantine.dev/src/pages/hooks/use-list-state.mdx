import { MDX_DATA } from '@/mdx'

export default Layout(MDX_DATA.useListState)

## Usage

The `use-list-state` hook provides an API to work with list state. It returns a `Ref<T[]>` and a
handlers object, similar to `useState` in React:

```vue
<script setup lang="ts">
import { useListState } from '@mantine-vue/hooks'

const [values, handlers] = useListState([{ a: 1 }])

// add one or more items to the end of the list
const append = () => handlers.append({ a: 2 })
// values.value -> [{ a: 1 }, { a: 2 }]

// add one or more items to the start of the list
const prepend = () => handlers.prepend({ a: 3 }, { a: 4 })
// values.value -> [{ a: 3 }, { a: 4 }, { a: 1 }, { a: 2 }]

// remove items at given positions
const remove = () => handlers.remove(0, 2)
// values.value -> [{ a: 4 }, { a: 2 }]

// insert one or more items at given position
const insert = () => handlers.insert(1, { a: 5 })
// values.value -> [{ a: 4 }, { a: 5 }, { a: 2 }]

// apply function to each element of the list
const apply = () => handlers.apply((item, index) => ({ a: item.a * (index ?? 0) }))
// values.value -> [{ a: 0 }, { a: 5 }, { a: 4 }]

// move item from one position to another
const reorder = () => handlers.reorder({ from: 2, to: 0 })
// values.value -> [{ a: 4 }, { a: 0 }, { a: 5 }]

// swap items positions
const swap = () => handlers.swap({ from: 0, to: 2 })
// values.value -> [{ a: 5 }, { a: 0 }, { a: 4 }]

// apply function to each element that matches condition
const applyWhere = () =>
  handlers.applyWhere(
    (item) => item.a > 0,
    (item) => ({ a: item.a + 2 }),
  )
// values.value -> [{ a: 7 }, { a: 0 }, { a: 6 }]

// set entirely new state
const setState = () => handlers.setState([{ a: 6 }, { a: 7 }])
// values.value -> [{ a: 6 }, { a: 7 }]

// set individual item at given position
const setItem = () => handlers.setItem(0, { a: 8 })
// values.value -> [{ a: 8 }, { a: 7 }]

// set item property at given position
const setItemProp = () => handlers.setItemProp(1, 'a', 'new-prop')
// values.value -> [{ a: 8 }, { a: 'new-prop' }]

// filter objects that have a = 'new-prop'
const filter = () => handlers.filter((item) => item.a === 'new-prop')
// values.value -> [{ a: 'new-prop' }]
</script>
```

## API

`use-list-state` takes an array (or a function returning an array) as a single argument and
returns a tuple of a `Ref<T[]>` and handlers to change its value, similar to the `useState` hook.

The hook provides handlers to work with array data:

- `append` – add items to the end of the list
- `prepend` – add items to the start of the list
- `pop` – remove last item
- `shift` – remove first item
- `insert` – insert items at given index
- `remove` – remove items at given indices
- `reorder` – move item from one position to another
- `swap` – swap items positions
- `apply` – apply given function to all items in the list
- `applyWhere` – apply given function to selective items using condition
- `setItem` – replace item at given index
- `setItemProp` – set item property at given index
- `setState` – set list state directly or with an updater function
- `filter` – filter values with callback function

## UseListStateHandlers type

`@mantine-vue/hooks` exports `UseListStateHandlers`. It is a generic type that contains all
handlers from the `useListState` hook. It can be used to type handlers in your components.

`UseListStateHandlers` type:

```ts
export interface UseListStateHandlers<T> {
  setState: (value: T[] | ((current: T[]) => T[])) => void
  append: (...items: T[]) => void
  prepend: (...items: T[]) => void
  insert: (index: number, ...items: T[]) => void
  pop: () => void
  shift: () => void
  apply: (fn: (item: T, index?: number) => T) => void
  applyWhere: (
    condition: (item: T, index: number) => boolean,
    fn: (item: T, index?: number) => T,
  ) => void
  remove: (...indices: number[]) => void
  reorder: (payload: { from: number; to: number }) => void
  swap: (payload: { from: number; to: number }) => void
  setItem: (index: number, item: T) => void
  setItemProp: <K extends keyof T, U extends T[K]>(index: number, prop: K, value: U) => void
  filter: (fn: (item: T, i: number) => boolean) => void
}
```

The type is useful when you want to pass `use-list-state` handlers to child components as a prop:

```vue
<script setup lang="ts">
import type { UseListStateHandlers } from '@mantine-vue/hooks'

const props = defineProps<{ handlers: UseListStateHandlers<string> }>()
</script>

<template>
  <button type="button" @click="handlers.append('hello')">Append hello</button>
</template>
```

## Set item type

By default, `use-list-state` will infer the item type from `initialValue`.
If you call the hook with an empty array, you must specify the item type explicitly:

```ts
import { useListState } from '@mantine-vue/hooks'

useListState(['hello']) // ok, item type is string
useListState([]) // not ok, item type is unknown
useListState<string>([]) // ok, item type is string
```

## Definition

```ts
function useListState<T>(initialValue?: T[] | (() => T[])): UseListStateReturnValue<T>

type UseListStateReturnValue<T> = [Ref<T[]>, UseListStateHandlers<T>]
```

## Exported types

`UseListStateReturnValue` and `UseListStateHandlers` types are exported from the `@mantine-vue/hooks`
package, you can import them in your application:

```ts
import type { UseListStateReturnValue, UseListStateHandlers } from '@mantine-vue/hooks'
```
