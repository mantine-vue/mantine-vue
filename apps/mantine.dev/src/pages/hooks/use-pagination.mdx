## Usage

The `use-pagination` hook is a state management hook for the
[Pagination](/core/pagination/) component; it manages pagination with
controlled and uncontrolled state:

```ts
import { usePagination } from '@mantine-vue/hooks'

const pagination = usePagination({ total: 10, initialPage: 1 })

pagination.range.value // -> [1, 2, 3, 4, 5, 'dots', 10]

pagination.setPage(5)
pagination.range.value // -> [1, 'dots', 4, 5, 6, 'dots', 10]

pagination.next()
pagination.range.value // -> [1, 'dots', 5, 6, 7, 'dots', 10]

pagination.previous()
pagination.range.value // -> [1, 'dots', 4, 5, 6, 'dots', 10]

pagination.last()
pagination.range.value // -> [1, 'dots', 6, 7, 8, 9, 10]

pagination.first()
pagination.range.value // -> [1, 2, 3, 4, 5, 'dots', 10]
```

`range` and `active` are returned as Vue refs, `range` is a computed ref
recalculated every time `active`, `total`, `siblings` or `boundaries` change –
access their current value with `.value` (or let it auto-unwrap in templates
via a top-level `const { range, active } = pagination` destructure).

## Controlled

The hook supports controlled mode; provide `page` and `onChange` options to
manage state from outside. `page` accepts a plain value, a `Ref`, or a getter
function (`MaybeRefOrGetter<number | undefined>`):

```ts
import { ref } from 'vue'
import { usePagination } from '@mantine-vue/hooks'

const page = ref(1)
const pagination = usePagination({
  total: 10,
  page,
  onChange: (value) => (page.value = value),
})

// Will call onChange with 5
pagination.setPage(5)
pagination.range.value // -> [1, 'dots', 4, 5, 6, 'dots', 10]

// ... All other examples work the same
```

## Siblings

Control the number of active item siblings with `siblings`:

```ts
import { usePagination } from '@mantine-vue/hooks'

const pagination = usePagination({ total: 20, siblings: 3 })
```

## Boundaries

Control the number of items on each boundary with `boundaries`:

```ts
import { usePagination } from '@mantine-vue/hooks'

const pagination = usePagination({ total: 20, boundaries: 3 })
```

## Start value

Set `startValue` to define the starting page number. For example, with
`startValue: 5` and `total: 15`, the pagination range will be from 5 to 15:

## DOTS

`@mantine-vue/hooks` exports the `DOTS` constant (`'dots'`), which is the value
used inside `range` to represent a collapsed gap of pages. Compare against it
instead of hardcoding the `'dots'` string literal:

```ts
import { DOTS, usePagination } from '@mantine-vue/hooks'

const pagination = usePagination({ total: 20 })

pagination.range.value.map((page) => (page === DOTS ? '…' : page))
```

## Definition

```ts
interface UsePaginationOptions {
  /** Page selected on initial render, defaults to 1 or startValue if provided */
  initialPage?: number

  /** Controlled active page number */
  page?: MaybeRefOrGetter<number | undefined>

  /** Total amount of pages */
  total: number

  /** Siblings amount on left/right side of selected page, defaults to 1 */
  siblings?: number

  /** Amount of elements visible on left/right edges, defaults to 1 */
  boundaries?: number

  /** Callback fired after change of each page */
  onChange?: (page: number) => void

  /** Starting page number, defaults to 1 */
  startValue?: number
}

interface UsePaginationReturnValue {
  /** Array of page numbers and dots */
  range: ComputedRef<(number | typeof DOTS)[]>

  /** Active page number */
  active: Ref<number>

  /** Function to set active page */
  setPage: (page: number) => void

  /** Function to go to next page */
  next: () => void

  /** Function to go to previous page */
  previous: () => void

  /** Function to go to first page */
  first: () => void

  /** Function to go to last page */
  last: () => void
}

function usePagination(options: UsePaginationOptions): UsePaginationReturnValue
```

## Exported types

`UsePaginationOptions` and `UsePaginationReturnValue` types are exported from
the `@mantine-vue/hooks` package; you can import them in your application:

```ts
import type { UsePaginationOptions, UsePaginationReturnValue } from '@mantine-vue/hooks'
```
