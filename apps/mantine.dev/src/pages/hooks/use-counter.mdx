## Definition

```ts
interface UseCounterOptions {
  min?: number
  max?: number
  step?: number
}

interface UseCounterHandlers {
  increment: () => void
  decrement: () => void
  set: (value: number | ((current: number) => number)) => void
  reset: () => void
}

type UseCounterReturnValue = [Ref<number>, UseCounterHandlers]

function useCounter(initialValue?: number, options?: UseCounterOptions): UseCounterReturnValue
```

## Exported types

The `UseCounterOptions`, `UseCounterHandlers`, and `UseCounterReturnValue` types are exported from the `@mantine-vue/hooks` package;
you can import them in your application:

```ts
import type {
  UseCounterOptions,
  UseCounterHandlers,
  UseCounterReturnValue,
} from '@mantine-vue/hooks'
```
